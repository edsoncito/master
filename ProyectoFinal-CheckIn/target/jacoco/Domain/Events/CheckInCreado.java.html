package Events;

import java.time.LocalDateTime;

    public String CodigoSeguridad;

    public CheckInCreado(UUID key, String codigoSeguridad) {
        super(LocalDateTime.now());
        KeyCheckIn = key;
        CodigoSeguridad = codigoSeguridad.toString();

}
